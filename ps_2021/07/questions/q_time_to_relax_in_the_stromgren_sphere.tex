\qns{Time to Relax in the Str\"{o}mgren Sphere}

\begin{enumerate}

\qitem{}
\work{
	\begin{align}
    	t_e &= \frac{\lambda_\text{mfp,e}}{v}\\
	        &= \frac{1}{n_e\sigma_ev}\\
	        &\approx \frac{1}{n_e\pi \frac{e^4}{m_e^2v^3}}  \longleftarrow
	            \begin{cases}
	                \sigma_e \approx \pi r_e^2\\
	                m_ev^2 \approx \frac{e^2}{r_e} \longleftarrow \text{Coulomb scattering}
	            \end{cases}\\
	        &\approx \frac{m_e^2\left(\frac{2k_BT_e}{m_e}\right)^\frac{3}{2}}{n_e\pi e^4}
	\end{align}}
\ans{
	\centering
	$$t_e \approx \frac{m_e^\frac{1}{2}\left(2k_BT_e\right)^\frac{3}{2}}{n\pi e^4}$$}

\qitem{\textcolor{white}{blep}}
\ans{
	\centering
	$$t_p \approx \frac{m_p^\frac{1}{2}\left(2k_BT_p\right)^\frac{3}{2}}{n\pi e^4}$$}

\qitem{}
\work{
	$t_\text{collision} = \frac{t_{ep}}{N}$ as the time for an electron to collide with a proton
	\begin{align*}
		\frac{1}{t_\text{collision}} &\approx \frac{1}{t_e} + \frac{1}{t_p}\\
		% t_\text{collision} &= \frac{t_et_p}{t_e + t_p}\\
		% 	&= \frac{(2k_B)^\frac{3}{2}}{n\pi e^4}\frac{(m_em_p)^\frac{1}{2}\left(T_pT_e\right)^\frac{3}{2}}{m_e^\frac{1}{2}T_e^\frac{3}{2} + m_p^\frac{1}{2}T_p^\frac{3}{2}}
	\end{align*}
	Finding the energy a proton picks up from a single collision $\Delta E_p$
	\begin{align*}
		\Delta E_p &= \frac{1}{2}m_p(v_{p,f}^2 - v_{p,i}^2)\\
			&\approx \frac{1}{2}m_pv_{p,f}^2 \longleftarrow T_p \text{ is low}\\
			&= \frac{1}{2}m_p\left(\frac{m_e}{m_p}\Delta v_e\right)^2 \longleftarrow m_e\Delta v_e = m_p\Delta v_p \approx m_pv_{p,f}\\
			&\approx \frac{1}{2}\frac{m_e^2}{m_p}(\Delta v_e)^2\\
			&\approx \frac{1}{2}\frac{m_e^2}{m_p}v_e^2 \longleftarrow \text{yay for order of magnitude}
	\end{align*}
	And comparing against the kinetic energy of a hot electron
	\begin{align*}
		t_{ep} &= t_\text{collision}\frac{\frac{1}{2}m_ev_e^2}{\Delta E_p}\\
			&\approx \frac{m_p}{m_e}t_\text{collision}
	\end{align*}}
\ans{
	\centering
	$$t_{ep} \approx \frac{m_p}{m_e}t_\text{collision},\text{ where }t_\text{collision} = \frac{t_et_p}{t_e+t_p}$$
}

\qitem{\textcolor{white}{bloop}}
\ans{
	\begin{align*}
		\frac{t_e}{t_\text{rec}} &\approx 3(10^{-7})\\
		\frac{t_p}{t_\text{rec}} &\approx 1(10^{-5})\\
		\frac{t_{ep}}{t_\text{rec}} &\approx 5(10^{-4})
	\end{align*}
	All collision times are significantly shorter than the recombination time, so assuming a Maxwellian distribution for elecrtrons and protons is safe.}

\end{enumerate}
