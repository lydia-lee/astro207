\qns{Hyperfine $\bm{^3}$He$\bm{^+}$}

\begin{enumerate}

\qitem{}
\work{
	\begin{align*}
		a_{0,\text{He}} &= \frac{0.52}{Z_\text{He}}\si{\angstrom}\\
		\text{Ryd}_\text{He} &= \frac{Z_\text{He}e^2}{a_{0,\text{He}}}\\
			&= \frac{Z_\text{He}^2e^2}{0.52}\\
			&= 4\cdot 13.6\si{\electronvolt}\\
			&= 54.4\si{\electronvolt}\\
		\text{Ly}\alpha_\text{He} &= \text{Ryd}_\text{He}\left(\frac{1}{n_1^2} - \frac{1}{n_2^2}\right)\\
			&= 4\cdot 10.2\si{\electronvolt}\\
			&= 40.8\si{\electronvolt}\\
		\lambda &= \frac{121.6\si{\nano\meter}}{4}\\
			&= 30.4\si{\nano\meter}
	\end{align*}}
\ans{
	$$\lambda \approx 30.4\si{\nano\meter}$$}

\qitem{
	\begin{itemize}
		\item nucleus charge: $Ze$, $Z=2$
		\item nucleus mass: $Cm_p$, $C=3$
	\end{itemize}}
\work{
	$$\Delta E \approx \mu_e \cdot B_\text{nucleus}$$
	\begin{align*}
		B_\text{nucleus,$^3$He} &\propto \frac{\mu_\text{nucleus,He}}{a_{0,\text{He}}^3}\\
			&= \frac{Ze\hbar}{2Cm_pc} \cdot \frac{1}{a_{0,\text{H}}^3/Z^3}\\
			&= B_\text{nucleus,H} \cdot \frac{Z^4}{C}\\
			&= B_\text{nucleus,H} \cdot \frac{16}{3}\\
		\lambda &\propto \frac{1}{E}\\
			&\propto \frac{1}{B_\text{nucleus}}\\
		\lambda_\text{He,hf} &= \frac{3}{16}\lambda_\text{H,hf}\\
			&\approx 3.94\si{\centi\meter}
	\end{align*}}
\ans{
	\centering
	$\lambda_\text{He,hf} \approx 3.94\si{\centi\meter}$
	Not too far off from 3.46\si{\centi\meter}}

\end{enumerate}
